# Use this file to suppress TSAN issues from 3rd party libraries
# https://github.com/google/sanitizers/wiki/ThreadSanitizerSuppressions

# vkDestroyDevice called in igl::vulkan::Device::~Device() causes false-positive in TSAN
race:igl::vulkan::Device::~Device()

# libvulkan_lvp.so (LLVMpipe Vulkan driver)
# Lavapipe has internal data races between render and worker threads
# (e.g. realloc vs memcpy, pthread_mutex ops). Suppress all races in
# the lavapipe shared library since we cannot fix third-party code.
race:libvulkan_lvp.so

# libLLVM.so (LLVM library used by Lavapipe)
# Lavapipe uses LLVM for JIT compilation, and LLVM has internal data races
# in StringMap operations between llvmpipe worker threads during shader
# compilation. This is third-party code we cannot fix.
race:libLLVM.so

# SwiftShader OpaqueFdExternalMemory uses an unsynchronized static counter
# for generating unique memfd names in allocateBuffer(). Benign race we cannot
# fix in third-party code.
race:OpaqueFdExternalMemory::allocateBuffer

# Khronos Vulkan-Loader (third-party, statically linked) has an unsynchronized
# read/write race on its process-global ICD device linked list when a new
# VkDevice is created on one thread while another thread is using a previously
# created VkDevice from the same ICD. This is benign at the application level
# (the two devices are independent) but TSAN flags loader.c:1681 (write in
# loader_add_logical_device) racing against loader.c:1639 (read in
# loader_get_icd_and_device). It surfaces in integration tests that create a
# client-side VkDevice (XrInstanceMock) while the server-side render thread is
# already issuing Vulkan calls on its own VkDevice. The unfixed race in the
# Khronos loader exits the process with signal 66 after the test passes,
# masking the test result as a fatal. Suppress here since we cannot patch
# third-party Vulkan-Loader code.
race:loader_add_logical_device
race:loader_get_icd_and_device
race:loader_remove_logical_device
